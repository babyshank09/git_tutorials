COMMANDS:

git --version                   # Tells you the version of git  

git init                        # Initializes a git directory in the present working directory 

git config --global user.name 
git config --global user.email 

git status 

git add README.txt              # README.txt is the file name. This command adds the file to the staging environment 

git commit -m "This is the first commit"   

git branch 

git branch -M main              # Renames the branch name to main 


git remote add origin https://github.com/babyshank09/git_tutorials.git 

git remote -v  

git push origin main  

git add .                       # Adds all the files in the directory  

git restore README.txt          # 

git reset
